81  xt::xtensor<double, Weibull::dim_param + Weibull::dim_stim + 1> generate_likelihoods()

87  const auto lower = xt::adapt<xt::layout_type::row_major>(settings.lower_asymptote, sz{1, 1, 1, settings.lower_asymptote.size(), 1});
88  const auto lapse = xt::adapt<xt::layout_type::row_major>(settings.lapse_rate, sz{1, 1, 1, 1, settings.lapse_rate.size()});
89 
90  xt::xtensor<double, Weibull::dim_param + Weibull::dim_stim> p;
91  switch (settings.stim_scale)
92  {
93  case Scale::Linear:
94  p = 1 - lapse - (1 - lower - lapse) * xt::exp(-xt::pow(x / thresh, slope));
95  break;
96  case Scale::Log10:
97  p = 1 - lapse - (1 - lower - lapse) * xt::exp(-xt::pow(10, slope * (x - thresh)));
98  break;
99  case Scale::dB:
100  p = 1 - lapse - (1 - lower - lapse) * xt::exp(-xt::pow(10, slope * (x - thresh) * 0.05));
101  break;
102  }
103  // in this, we diverge from hoechenberger/questplus
104  // store 0/incorrect as 0th element, so that we can index using the response
105  return xt::stack(xt::xtuple(1.0 - p, p));
106  }
