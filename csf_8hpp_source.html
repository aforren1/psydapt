99  xt::xtensor<double, CSF::dim_param + CSF::dim_stim + 1> generate_likelihoods()
100  {
101  // (7 param, 3 stim)
102  const Params &set = settings;
103  using sz = std::array<std::size_t, CSF::dim_param + CSF::dim_stim>;
104  // stim
105  auto x = xt::adapt(set.contrast, sz{set.contrast.size(), 1, 1, 1, 1, 1, 1, 1, 1, 1});
106  auto f = xt::adapt(set.spatial_freq, sz{1, set.spatial_freq.size(), 1, 1, 1, 1, 1, 1, 1, 1});
107  auto w = xt::adapt(set.temporal_freq, sz{1, 1, set.temporal_freq.size(), 1, 1, 1, 1, 1, 1, 1});
108  // param
109  auto c0 = xt::adapt(set.c0, sz{1, 1, 1, set.c0.size(), 1, 1, 1, 1, 1, 1});
110  auto cf = xt::adapt(set.cf, sz{1, 1, 1, 1, set.cf.size(), 1, 1, 1, 1, 1});
111  auto cw = xt::adapt(set.cw, sz{1, 1, 1, 1, 1, set.cw.size(), 1, 1, 1, 1});
112  auto min_thresh = xt::adapt(set.min_thresh, sz{1, 1, 1, 1, 1, 1, set.min_thresh.size(), 1, 1, 1});
113  auto slope = xt::adapt(set.slope, sz{1, 1, 1, 1, 1, 1, 1, set.slope.size(), 1, 1});
114  auto lower = xt::adapt(set.lower_asymptote, sz{1, 1, 1, 1, 1, 1, 1, 1, set.lower_asymptote.size(), 1});
115  auto lapse = xt::adapt(set.lapse_rate, sz{1, 1, 1, 1, 1, 1, 1, 1, 1, set.lapse_rate.size()});
116 
117  auto t = xt::maximum(min_thresh, c0 + cf * f + cw * w);
118  xt::xtensor<double, CSF::dim_param + CSF::dim_stim> p;
119  switch (settings.stim_scale)
120  {
121  case Scale::Linear:
122  p = 1 - lapse - (1 - lower - lapse) * xt::exp(-xt::pow(x / t, slope));
123  break;
124  case Scale::Log10:
125  p = 1 - lapse - (1 - lower - lapse) * xt::exp(-xt::pow(10, slope * (x - t)));
126  break;
127  case Scale::dB:
128  p = 1 - lapse - (1 - lower - lapse) * xt::exp(-xt::pow(10, slope * (x - t) / 20.0));
129  break;
130  }
131  // in this, we diverge from hoechenberger/questplus
132  // store 0/incorrect as 0th element, so that we can index using the response
133  return xt::stack(xt::xtuple(1.0 - p, p));
134  }
