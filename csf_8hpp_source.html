84 xt::xtensor<double, CSF::dim_param> generate_prior()
85 {
86 const auto c0_prior = prior_helper(settings.c0, settings.c0_prior, 0);
87 const auto cf_prior = prior_helper(settings.cf, settings.cf_prior, 1);
88 const auto cw_prior = prior_helper(settings.cw, settings.cw_prior, 2);
89 const auto min_thresh_prior = prior_helper(settings.min_thresh, settings.min_thresh_prior, 3);
90 const auto slope_prior = prior_helper(settings.slope, settings.slope_prior, 4);
91 const auto lower_prior = prior_helper(settings.lower_asymptote, settings.lower_asymptote_prior, 5);
92 const auto lapse_prior = prior_helper(settings.lapse_rate, settings.lapse_rate_prior, 6);
93 xt::xtensor<double, CSF::dim_param> prior = c0_prior * cf_prior * cw_prior * min_thresh_prior * slope_prior * lower_prior * lapse_prior;
94 return prior / xt::sum(prior, xt::evaluation_strategy::immediate);
95 }

97 xt::xtensor<double, CSF::dim_param + CSF::dim_stim + 1> generate_likelihoods()
98 {
99 // (7 param, 3 stim)
100 const Params &set = settings;
101 using sz = std::array<std::size_t, CSF::dim_param + CSF::dim_stim>;
102 // const auto &row_major = xt::layout_type::row_major;
103 // stim
104 const auto x = xt::adapt<xt::layout_type::row_major>(set.contrast, sz{set.contrast.size(), 1, 1, 1, 1, 1, 1, 1, 1, 1});
105 const auto f = xt::adapt<xt::layout_type::row_major>(set.spatial_freq, sz{1, set.spatial_freq.size(), 1, 1, 1, 1, 1, 1, 1, 1});
106 const auto w = xt::adapt<xt::layout_type::row_major>(set.temporal_freq, sz{1, 1, set.temporal_freq.size(), 1, 1, 1, 1, 1, 1, 1});
107 // param
108 const auto c0 = xt::adapt<xt::layout_type::row_major>(set.c0, sz{1, 1, 1, set.c0.size(), 1, 1, 1, 1, 1, 1});
109 const auto cf = xt::adapt<xt::layout_type::row_major>(set.cf, sz{1, 1, 1, 1, set.cf.size(), 1, 1, 1, 1, 1});
110 const auto cw = xt::adapt<xt::layout_type::row_major>(set.cw, sz{1, 1, 1, 1, 1, set.cw.size(), 1, 1, 1, 1});
111 const auto min_thresh = xt::adapt<xt::layout_type::row_major>(set.min_thresh, sz{1, 1, 1, 1, 1, 1, set.min_thresh.size(), 1, 1, 1});
112 const auto slope = xt::adapt<xt::layout_type::row_major>(set.slope, sz{1, 1, 1, 1, 1, 1, 1, set.slope.size(), 1, 1});
113 const auto lower = xt::adapt<xt::layout_type::row_major>(set.lower_asymptote, sz{1, 1, 1, 1, 1, 1, 1, 1, set.lower_asymptote.size(), 1});
114 const auto lapse = xt::adapt<xt::layout_type::row_major>(set.lapse_rate, sz{1, 1, 1, 1, 1, 1, 1, 1, 1, set.lapse_rate.size()});
115
116 const auto t = xt::maximum(min_thresh, c0 + cf * f + cw * w);
117 xt::xtensor<double, CSF::dim_param + CSF::dim_stim> p;
118 switch (settings.stim_scale)
119 {
120 case Scale::Linear:
121 p = 1 - lapse - (1 - lower - lapse) * xt::exp(-xt::pow(x / t, slope));
122 break;
123 case Scale::Log10:
124 p = 1 - lapse - (1 - lower - lapse) * xt::exp(-xt::pow(10, slope * (x - t)));
125 break;
126 case Scale::dB:
127 p = 1 - lapse - (1 - lower - lapse) * xt::exp(-xt::pow(10, slope * (x - t) * 0.05));
128 break;
129 }
130 // in this, we diverge from hoechenberger/questplus
131 // store 0/incorrect as 0th element, so that we can index using the response
132 return xt::stack(xt::xtuple(1.0 - p, p));
133 }
