43 template <typename T>
44 T norm_cdf(T x)
45 {
46 return std::erfc(-x * std::sqrt(0.5)) * 0.5;
47 }
48 static auto vec_norm_cdf = xt::vectorize(norm_cdf<double>);
49 } // namespace detail

82 xt::xtensor<double, NormCDF::dim_param> generate_prior()
83 {
84 const auto loc_prior = prior_helper(settings.location, settings.location_prior, 0);
85 const auto scale_prior = prior_helper(settings.scale, settings.scale_prior, 1);
86 const auto lower_prior = prior_helper(settings.lower_asymptote, settings.lower_asymptote_prior, 2);
87 const auto lapse_prior = prior_helper(settings.lapse_rate, settings.lapse_rate_prior, 3);
88 xt::xtensor<double, NormCDF::dim_param> prior = loc_prior * scale_prior * lower_prior * lapse_prior;
89 return prior / xt::sum(prior, xt::evaluation_strategy::immediate);
90 }
91
92 xt::xtensor<double, NormCDF::dim_param + NormCDF::dim_stim + 1> generate_likelihoods()
93 {
94 using sz = std::array<std::size_t, NormCDF::dim_param + NormCDF::dim_stim>;
95 const auto x = xt::adapt<xt::layout_type::row_major>(settings.intensity, sz{settings.intensity.size(), 1, 1, 1, 1});
96 const auto loc = xt::adapt<xt::layout_type::row_major>(settings.location, sz{1, settings.location.size(), 1, 1, 1});
97 const auto scale = xt::adapt<xt::layout_type::row_major>(settings.scale, sz{1, 1, settings.scale.size(), 1, 1});
98 const auto lower = xt::adapt<xt::layout_type::row_major>(settings.lower_asymptote, sz{1, 1, 1, settings.lower_asymptote.size(), 1});
99 const auto lapse = xt::adapt<xt::layout_type::row_major>(settings.lapse_rate, sz{1, 1, 1, 1, settings.lapse_rate.size()});
100
101 xt::xtensor<double, NormCDF::dim_param + NormCDF::dim_stim> p;
102 switch (settings.stim_scale)
103 {
104 case Scale::Linear:
105 p = lower + (1 - lower - lapse) * detail::vec_norm_cdf((x - loc) / scale);
106 break;
107 default:
108 PSYDAPT_THROW(std::invalid_argument, "Only 'Linear' stimulus scale is implemented for NormCDF.");
109 break;
110 }
111 return xt::stack(xt::xtuple(1.0 - p, p));
112 }
