82  stim_type next()
83  {
84  xt::xtensor<double, (DimParam + DimStim + 1)> new_posterior = posterior * likelihoods;
85  std::array<std::size_t, DimParam> param_idx;
86  std::iota(param_idx.begin(), param_idx.end(), 1 + DimStim);
87  auto pk = xt::sum(new_posterior, param_idx, xt::evaluation_strategy::immediate);
88  xt::transpose(new_posterior) /= xt::transpose(pk);
89 
90  // entropy
91  auto H = -xt::nansum((new_posterior * xt::log(new_posterior)), param_idx,
92  xt::evaluation_strategy::immediate);
93  // expected entropies for stimuli
94  auto EH = xt::sum(pk * H, 0, xt::evaluation_strategy::immediate);
95  // TODO: just do min_entropy by default until figure out retrieving settings
96  // find index of minimum entropy, then figure out which stimuli are there
97  if constexpr (std::is_scalar_v<stim_type>)
98  {
99  this->next_stimulus = stimuli[0][xt::argmin(EH)];
100  }
101  else
102  {
103  auto indices = xt::unravel_index(xt::argmin(EH)(), EH.shape(), xt::layout_type::row_major);
104  stim_type out;
105  for (std::size_t i = 0; i < DimStim; i++)
106  {
107  out[i] = stimuli[i][indices[i]];
108  this->next_stimulus[i] = out[i];
109  }
110  }
111  return this->next_stimulus;
112  }
113  bool update(unsigned int response, std::optional<stim_type> stimulus = std::nullopt)
114  {
115  if (response >= n_resp)
116  {
117  using namespace std::string_literals;
118  throw std::invalid_argument("The response " + std::to_string(response) + " was not within [0, " + std::to_string(n_resp) + ")."s);
119  }
120  this->stimulus_history.push_back(stimulus ? *stimulus : this->next_stimulus);
121  this->response_history.push_back(response);
122  // find nearest matching response
123  xt::xtensor<double, DimParam + DimStim> likelihood2 = xt::view(likelihoods, response, xt::all());
124  const auto last_stim = this->stimulus_history.back();
125 
126  xt::xtensor<double, DimParam> likelihood;
127  if constexpr (std::is_scalar_v<stim_type>)
128  {
129  // find index of nearest stimulus input, and take a view from there
130  std::size_t idx = xt::argmin(xt::abs(stimuli[0] - last_stim))[0];
131  likelihood = xt::view(likelihood2, idx, xt::all());
132  }
133  else
134  {
135  // need to loop over all of DimStim/whatever the stim length is,
136  // taking progressively more views
137  std::array<std::size_t, DimStim> idx;
138  for (std::size_t i = 0; i < DimStim; i++)
139  {
140  idx[i] = xt::argmin(xt::abs(stimuli[i] - last_stim[i]))[0];
141  }
142  likelihood = std::apply([likelihood2](auto &&...xs) { return xt::view(likelihood2, xs...); }, idx);
143  }
144  posterior *= likelihood;
145  posterior /= xt::sum(posterior, xt::evaluation_strategy::immediate);
146 
147  return true; // unconditionally continue for now
148  }
